\begin{frame}{Key Process Reactions}
\textbf{Cr/Mn Acid Leach} (80\textdegree C, 2~hr residence):
\[
\text{Cr}_2\text{O}_3 + 6\,\text{HCl} \to 2\,\text{CrCl}_3 + 3\,\text{H}_2\text{O}
\]
\[
\text{MnO} + 2\,\text{HCl} \to \text{MnCl}_2 + \text{H}_2\text{O}
\]

\textbf{Selective Precipitation:}
\begin{itemize}
\item Cr(OH)\textsubscript{3} at pH~8.5 (NaOH, 70\% recovery)
\item Mn(OH)\textsubscript{2} at pH~9.5 (NaOH, 65\% recovery)
\end{itemize}

\textbf{V Acid Leach} (90\textdegree C, 3~hr) + NH\textsubscript{4}Cl precipitation (60\% recovery)
\end{frame}
