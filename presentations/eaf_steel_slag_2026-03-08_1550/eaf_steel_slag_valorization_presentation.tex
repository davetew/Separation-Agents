% ═══ Key reactions ═══
\begin{frame}{Key Process Reactions (Equilibrium)}
\textbf{Cr Acid Leach} (80\textdegree C, 2~hr --- Gibbs minimization):
\[
\text{Cr}_2\text{O}_3 + 6\,\text{HCl} \to 2\,\text{CrCl}_3 + 3\,\text{H}_2\text{O}
\]
\centering \textit{Equilibrium: 99.4\% dissolution (Eskolaite)}

\textbf{Selective Cr Precipitation} (NaOH, pH 8.5):
\begin{itemize}
\item Eskolaite (Cr\textsubscript{2}O\textsubscript{3}) --- \textbf{99.9\% recovery at equilibrium}
\item Mn precipitation: \textcolor{StepRed}{zero extent} (Bixbyite/Manganosite infeasible at 50\textdegree C)
\end{itemize}

\textbf{V Recovery}: \textcolor{StepRed}{Not feasible} --- no V solid phases in SUPCRTBL database
\end{frame}
