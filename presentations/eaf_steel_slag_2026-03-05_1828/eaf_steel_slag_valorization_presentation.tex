\section{Background \& Opportunity}

\begin{frame}{Electric Arc Furnace Steel Slag}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\begin{itemize}
  \item EAF steelmaking generates 150--200 kg slag per tonne steel
  \item Global EAF production: $>$500 Mt/yr
  \item Higher Fe and Cr content than BOF slag
  \item Currently landfilled or low-value aggregate
\end{itemize}
\vspace{8pt}
\textbf{Opportunity:} Multi-product valorization targeting:
\begin{enumerate}
  \item Iron + Cr + Mn magnetic recovery
  \item Green H$_2$ via serpentinization
  \item CO$_2$ credits via carbonation

\begin{frame}{Key Reactions}
\textbf{Serpentinization} (H$_2$ production):
\begin{equation*}
3\,\text{Fe}_2\text{SiO}_4 + 2\,\text{H}_2\text{O}
\longrightarrow 2\,\text{Fe}_3\text{O}_4 + 3\,\text{SiO}_2 + 2\,\text{H}_2
\end{equation*}
\begin{itemize}
  \item Equilibrium conversion: $\sim$49\% at 200\textdegree C, 300 bar
  \item Modeled by Gibbs energy minimization (Reaktoro)
\end{itemize}

\vspace{10pt}
\textbf{Carbonation} (CO$_2$ mineralization):
\begin{equation*}
\text{CaO} + \text{CO}_2 \rightarrow \text{CaCO}_3
\qquad\qquad
\text{MgO} + \text{CO}_2 \rightarrow \text{MgCO}_3
\end{equation*}
\begin{itemize}
  \item Equilibrium conversion: $\sim$100\% at 185\textdegree C, 150 bar
  \item 323 kg CO$_2$ sequestered per tonne slag
\end{itemize}
\end{frame}

\begin{frame}{Key Sensitivities}
\begin{itemize}
  \item \textbf{Cr$_2$O$_3$ price $\pm$30\%}: $\Delta$Revenue $= \pm$\$1.0M/yr
  \item \textbf{CO$_2$ credits to \$80/t}: +\$0.65M/yr (EU ETS pricing)
  \item \textbf{CAPEX $-$20\%}: NPV improves by \$8.5M
  \item \textbf{Discount rate 5\%}: NPV improves by $\sim$\$12M
  \item \textbf{H$_2$ price to \$6/kg}: +\$0.23M/yr (modest impact)
\end{itemize}

\vspace{8pt}
$\Rightarrow$ \textbf{Cr and CO$_2$ prices are the primary economic levers}

\section{Conclusions}

\begin{frame}{Key Findings}
\begin{enumerate}
  \item EAF slag valorization generates \textbf{\$9.4M/yr revenue} from 6 streams
  \item \textbf{Cr$_2$O$_3$ recovery} is the dominant value (\$3.4M/yr, 36\%)\\
        {\small despite being only 2\,wt\% of slag}
  \item CO$_2$ mineralization sequesters \textbf{32,280 t CO$_2$/yr}
  \item Net value: \textcolor{StepGreen}{\textbf{\$13.33/t slag}} (all 6 streams are positive)
  \item Project is \textbf{cash-flow positive} but CAPEX-intensive (NPV negative at 8\%)
\end{enumerate}

\vspace{10pt}

  \item Pilot-scale validation of serpentinization kinetics
