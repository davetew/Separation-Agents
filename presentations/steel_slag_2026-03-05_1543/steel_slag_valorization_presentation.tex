\section{Background \& Opportunity}

\begin{frame}{The Steel Slag Challenge}
\framesub{200+ Mt/yr generated globally, much landfilled}

\begin{itemize}
  \item BOF steel slag: primary by-product of steelmaking (\SI{100}{}-\SI{200}{kg/t} steel)
  \item Rich in \textbf{CaO} (40\%), \textbf{FeO} (20\%), \textbf{SiO\textsubscript{2}} (15\%), \textbf{MgO} (8\%)
  \item Minor: MnO (5\%), Cr\textsubscript{2}O\textsubscript{3} (2\%), V\textsubscript{2}O\textsubscript{5} (0.5\%)

\end{itemize}

\vspace{10pt}
\textbf{Study scope:} \SI{500}{t/d} BOF slag $\rightarrow$ H\textsubscript{2} + CaCO\textsubscript{3} + Fe concentrate
\end{frame}

\begin{frame}{Key Reactions}
\framesub{Serpentinization and mineral carbonation}

\textbf{H\textsubscript{2} Production (Serpentinization):}
\begin{equation*}
  3\text{Fe}_2\text{SiO}_4 + 2\text{H}_2\text{O}
  \longrightarrow 2\text{Fe}_3\text{O}_4 + 3\text{SiO}_2 + 2\text{H}_2 \uparrow
\end{equation*}

\vspace{12pt}
\textbf{CO\textsubscript{2} Sequestration (Mineral Carbonation):}
\begin{equation*}
  \text{CaO} + \text{CO}_2 \longrightarrow \text{CaCO}_3
  \qquad\quad
  \text{MgO} + \text{CO}_2 \longrightarrow \text{MgCO}_3
\end{equation*}

\vspace{12pt}
\begin{itemize}
  \item Serpentinization at \SI{300}{\celsius}, \SI{50}{atm} --- produces magnetite + H\textsubscript{2}
  \item Carbonation at \SI{180}{\celsius}, \SI{30}{atm} --- permanent CO\textsubscript{2} storage
\end{itemize}
\end{frame}

\section{Conclusions}
% ═══════════════════════════════════════════════════════════

\begin{frame}{Key Findings}

\begin{enumerate}
  \item BOF slag at 500 t/d produces \textbf{168 t/yr H\textsubscript{2}}, sequesters \textbf{37,884 t/yr CO\textsubscript{2}}, recovers \textbf{23,100 t/yr Fe}
  \item Optimal topology (Config-4: minimal) yields net \textbf{$-$\$6.83/t} at current prices
  \item \textbf{Acid consumption} is the dominant cost driver (74\% of OPEX)
  \item Breakeven at CO\textsubscript{2} $\geq$\SI{90}{\$/t} or acid $\leq$\SI{100}{kg/t}
  \item \textcolor{StepGreen}{Optimistic scenario: \textbf{+\$9.85/t} profit (\$1.63M/yr)}
\end{enumerate}

\vspace{12pt}
\textbf{Next steps:}
\begin{itemize}
  \item Pilot-scale serpentinization trials
  \item Acid regeneration feasibility study
  \item Include V/Cr co-product economics
\end{itemize}
\end{frame}
